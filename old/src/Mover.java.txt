import java.awt.*;
import javax.swing.JPanel;
import java.lang.Math;
import java.util.*;
import java.io.*;

class Mover {

    int frameCount = 0;
  
    boolean[][] state;
  
    int gridSize;
    int max;
    int increment;
  
    public Mover() {
      gridSize = 20;
      increment = 4;
      max = 400;
      state = new boolean[20][20];
      for (int i = 0; i < 20; i++) {
        for (int j = 0; j < 20; j++) {
          state[i][j] = false;
        }
      }
    }
  
    public void updateState(boolean[][] state) {
      for (int i = 0; i < 20; i++) {
        for (int j = 0; j < 20; j++) {
          this.state[i][j] = state[i][j];
        }
      }
    }
  
    public boolean isValidDest(int x, int y) {
  
      if ((((x) % 20 == 0) || ((y) % 20) == 0) && 20 <= x && x < 400 && 20 <= y && y < 400 && state[x / 20 - 1][y / 20 - 1]) {
        return true;
      }
      return false;
    }
  }
